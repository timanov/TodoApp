create table user(
id int PRIMARY KEY AUTO_INCREMENT,
username varchar(30),
password varchar(30),
firstName varchar(30),
lastName varchar(30),
middleName varchar(30),
email varchar(30),
enabled boolean
)

create table role(
id int PRIMARY KEY AUTO_INCREMENT,
name varchar(30)
)

create table user_role(
user_id int,
role_id int,
FOREIGN KEY (user_id)
REFERENCES user(id),
FOREIGN KEY (role_id)
REFERENCES role(id)
)

create table course(
id int PRIMARY KEY AUTO_INCREMENT,
name varchar(30)
)

create table lecture(
id int PRIMARY KEY AUTO_INCREMENT,
name varchar(30),
description varchar(30),
course_id int,
FOREIGN KEY (course_id)
REFERENCES course(id)
)
